Map<String, dynamic> createMessageRecordData({
  DocumentReference? chat,
  DocumentReference? sender,
  DateTime? datetime,
  String? message,
  bool? seen,
}) {
  final firestoreData = serializers.toFirestore(
    MessageRecord.serializer,
    MessageRecord(
      (m) => m
        ..chat = chat
        ..sender = sender
        ..datetime = datetime
        ..message = message
        ..seen = seen,
    ),
  );

  return firestoreData;
}
